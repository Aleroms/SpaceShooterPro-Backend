Space Shooter Pro API Documentation

User Authentication and Management

/login

Method: POST

Description: Logs in a user.

Request Body:

    {
        "username": "your_username",
        "password": "your_password"
    }

Response:

    {
        "response": {
            "success": "Successfully logged in",
            "session_token": "your_session_token"
        }
    }

Error Response:

    {
        "error": "Invalid username or password"
    }

/register

Method: POST

Description: Registers a new user.

Request Body:

    {
        "username": "your_username",
        "password": "your_password"
    }

Response:

    {
        "response": {
            "success": "Successfully added new user",
            "session_token": "your_session_token"
        }
    }

Error Response:

    {
        "error": {
            "Missing Field": "Username and password are required"
        }
    }
    
    {
        "error": {
            "Duplicate Entry": "User already exists"
        }
    }

/delete_user

Method: DELETE

Description: Deletes the currently logged-in user.

Headers:

    {
        "Authorization": "your_session_token"
    }

Response:

    {
        "response": {
            "success": "successfully deleted your_username"
        }
    }

Error Response:

    {
        "error": "Session token missing" 
    }
    
    {
        "error": "Invalid session token" 
    }

/logout

Method: POST

Description: Logs out the currently logged-in user.

Headers:

    {
        "Authorization": "your_session_token"
    }

Response:

    {
        "response": {
            "success": "successfully logged out"
        }
    }

Error Response:

    {
        "error": "Session token missing" 
    }
    
    {
        "error": "Invalid session token" 
    }

Game Logic

/update_highscore

Method: POST

Description: Updates the high score for the currently logged-in user.

Headers:

    {
        "Authorization": "your_session_token"
    }

Request Body:

    {
        "score": "your_new_score" 
    }

Response:

    {
        "response": {
            "success": "successfully updated the user's score"
        }
    }

Error Response:

    {
        "response": {
            "error": "there was an error with the score provided. Please check"
        }
    }
    
    {
        "response": {
            "error": "score is not new high score"
        }
    }
    
    {
        "error": "Session token missing" 
    }
    
    {
        "error": "Invalid session token" 
    }
